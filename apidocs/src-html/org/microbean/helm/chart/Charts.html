137  public static final Future<Release> install(final Tiller tiller,
138                                              final URL chartUrl,
139                                              final boolean disableHooks,
140                                              final boolean dryRun,
141                                              final String releaseName,
142                                              final String releaseNamespace,
143                                              final boolean reuseReleaseName,
144                                              final long timeoutInSeconds,
145                                              final String yamlValues,
146                                              final boolean wait)
147    throws IOException {
148    Objects.requireNonNull(tiller);
149    Objects.requireNonNull(chartUrl);
150    final Chart chart = new URLChartLoader().load(chartUrl);
151    final InstallReleaseRequest.Builder requestBuilder = InstallReleaseRequest.newBuilder();
152    assert requestBuilder != null;
153    requestBuilder.setChart(chart);
154    requestBuilder.setDisableHooks(disableHooks);
155    requestBuilder.setDryRun(dryRun);
156    if (releaseName != null && !releaseName.isEmpty()) {
157      requestBuilder.setName(releaseName);
158    }
159    if (releaseNamespace != null && !releaseNamespace.isEmpty()) {
160      requestBuilder.setNamespace(releaseNamespace);
161    } else {
162      requestBuilder.setNamespace("default"); // TODO: harvest from equivalent of kube config
163    }
164    requestBuilder.setReuseName(reuseReleaseName);
165    requestBuilder.setTimeout(timeoutInSeconds);
166    final Config.Builder configBuilder = Config.newBuilder();
167    assert configBuilder != null;
168    if (yamlValues != null && !yamlValues.isEmpty()) {
169      final ByteString rawBytes = ByteString.copyFrom(yamlValues, StandardCharsets.UTF_8);
170      assert rawBytes != null;
171      configBuilder.setRawBytes(rawBytes);
172    }
173    requestBuilder.setValues(configBuilder.build());
174    requestBuilder.setWait(wait);
175    final InstallReleaseRequest request = requestBuilder.build();
176    assert request != null;
177    final ReleaseServiceFutureStub stub = tiller.getReleaseServiceFutureStub();
178    assert stub != null;
179    final Future<InstallReleaseResponse> responseFuture = stub.installRelease(request);
180    final FutureTask<Release> returnValue;
181    if (responseFuture == null) {
182      returnValue = new FutureTask<>(() -> {}, null);
183    } else {
184      returnValue = new FutureTask<>(() -> {
185          final Release rv;
186          final InstallReleaseResponse response = responseFuture.get();
187          if (response == null) {
188            rv = null;
189          } else {
190            rv = response.getRelease();
191          }
192          return rv;
193        });
194    }
195    returnValue.run();
196    return returnValue;
197  }
