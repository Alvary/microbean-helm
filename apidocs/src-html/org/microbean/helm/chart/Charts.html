059  /**
060   * Using a {@link Tiller} constructed just in time for communication
061   * with the Tiller component of the Helm ecosystem, installs the
062   * Helm chart located at the supplied {@link URL} with default
063   * options.
064   *
065   * <p>This method never returns {@code null}.</p>
066   *
067   * @param chartUrl a {@link URL} that designates a Helm chart; must
068   * not be {@code null}; will be loaded with a {@link URLChartLoader}
069   *
070   * @return a {@link Future} whose {@link Future#get()} method will
071   * return the {@link Release} representing the chart that was
072   * installed
073   *
074   * @see #install(Tiller, URL, boolean, boolean, String, String,
075   * boolean, long, String, boolean)
076   *
077   * @see Tiller
078   *
079   * @exception IOException if a communication error occurs
080   *
081   * @exception KubernetesClientException if there is a problem with
082   * Kubernetes communication itself
083   *
084   * @exception NullPointerException if either {@code tiller} or
085   * {@code chartUrl} is {@code null}
086   */
087  public static final Future<Release> install(final URL chartUrl)
088    throws IOException {
089    try (final Tiller tiller = new Tiller(new DefaultKubernetesClient())) {
090      return install(tiller,
091                     chartUrl,
092                     false,
093                     false,
094                     null,
095                     null,
096                     false,
097                     300L, // See https://github.com/kubernetes/helm/blob/v2.5.0/cmd/helm/install.go#L189
098                     null,
099                     false);
100    }
101  }

337  /**
338   * Using the supplied {@link Tiller} for communication with the
339   * Tiller component of the Helm ecosystem, installs the Helm chart
340   * located at the supplied {@link URL} with the supplied options.
341   *
342   * <p>This method never returns {@code null}.</p>
343   *
344   * @param tiller a {@link Tiller} instance to use to communicate
345   * with Tiller; must not be {@code null}
346   *
347   * @param chartUrl a {@link URL} that designates a Helm chart; must
348   * not be {@code null}; will be loaded with a {@link URLChartLoader}
349   *
350   * @param disableHooks whether the chart's hooks should be disabled
351   *
352   * @param dryRun if {@code true} then no installation will take
353   * place, and a {@link Future} whose {@link Future#get()} method may
354   * return {@code null} will be returned
355   *
356   * @param releaseName the name of the release; if {@code null} or
357   * {@linkplain String#isEmpty() empty} then a generated name will be
358   * used instead
359   *
360   * @param releaseNamespace the namespace into which the release
361   * should be installed; if {@code null} or {@linkplain
362   * String#isEmpty() empty} {@code default} will be used instead
363   *
364   * @param reuseReleaseName whether to silently replace any extant
365   * release with the same {@code releaseName} (not suitable for
366   * production)
367   *
368   * @param timeoutInSeconds the number of seconds after which any
369   * Kubernetes operations will time out; <a
370   * href="https://github.com/kubernetes/helm/blob/v2.5.0/cmd/helm/install.go#L189">set
371   * in the Helm project to {@code 300L} by default</a>
372   *
373   * @param yamlValues a YAML-formatted {@link String} representing
374   * values to override in the chart; may be {@code null}
375   *
376   * @param wait whether to wait for Pods and Services and such to
377   * become Ready before the returned {@link Future}'s {@link
378   * Future#get()} method will return a {@link Release}
379   *
380   * @return a {@link Future} whose {@link Future#get()} method will
381   * return the {@link Release} representing the chart that was
382   * installed
383   *
384   * @see Tiller
385   *
386   * @exception IOException if a communication error occurs
387   *
388   * @exception KubernetesClientException if there is a problem with
389   * Kubernetes communication itself
390   *
391   * @exception NullPointerException if either {@code tiller} or
392   * {@code chartUrl} is {@code null}
393   */
394  public static final Future<Release> install(final Tiller tiller,
395                                              final URL chartUrl,
396                                              final boolean disableHooks,
397                                              final boolean dryRun,
398                                              final String releaseName,
399                                              final String releaseNamespace,
400                                              final boolean reuseReleaseName,
401                                              final long timeoutInSeconds,
402                                              final String yamlValues,
403                                              final boolean wait)
404    throws IOException {
405    Objects.requireNonNull(tiller);
406    Objects.requireNonNull(chartUrl);
407    Chart chart = null;
408    try (final ChartLoader<URL> loader = new URLChartLoader()) {      
409      chart = loader.load(chartUrl);
410    }
411    final InstallReleaseRequest.Builder requestBuilder = InstallReleaseRequest.newBuilder();
412    assert requestBuilder != null;
413    requestBuilder.setChart(chart);
414    requestBuilder.setDisableHooks(disableHooks);
415    requestBuilder.setDryRun(dryRun);
416    if (releaseName != null && !releaseName.isEmpty()) {
417      requestBuilder.setName(releaseName);
418    }
419    if (releaseNamespace != null && !releaseNamespace.isEmpty()) {
420      requestBuilder.setNamespace(releaseNamespace);
421    } else {
422      requestBuilder.setNamespace("default"); // TODO: harvest from equivalent of kube config
423    }
424    requestBuilder.setReuseName(reuseReleaseName);
425    requestBuilder.setTimeout(timeoutInSeconds);
426    final Config.Builder configBuilder = Config.newBuilder();
427    assert configBuilder != null;
428    if (yamlValues != null && !yamlValues.isEmpty()) {
429      final ByteString rawBytes = ByteString.copyFrom(yamlValues, StandardCharsets.UTF_8);
430      assert rawBytes != null;
431      configBuilder.setRawBytes(rawBytes);
432    }
433    requestBuilder.setValues(configBuilder.build());
434    requestBuilder.setWait(wait);
435    final InstallReleaseRequest request = requestBuilder.build();
436    assert request != null;
437    final ReleaseServiceFutureStub stub = tiller.getReleaseServiceFutureStub();
438    assert stub != null;
439    final Future<InstallReleaseResponse> responseFuture = stub.installRelease(request);
440    final FutureTask<Release> returnValue;
441    if (responseFuture == null) {
442      returnValue = new FutureTask<>(() -> {}, null);
443    } else {
444      returnValue = new FutureTask<>(() -> {
445          final Release rv;
446          final InstallReleaseResponse response = responseFuture.get();
447          if (response == null) {
448            rv = null;
449          } else {
450            rv = response.getRelease();
451          }
452          return rv;
453        });
454    }
455    returnValue.run();
456    return returnValue;
457  }
