017package org.microbean.helm.chart;
018
019import java.io.IOException;
020import java.io.InputStream;
021
022import java.util.Map;
023
024import com.google.protobuf.Any;
025import com.google.protobuf.ByteString;
026
027import hapi.chart.ChartOuterClass.Chart;
028import hapi.chart.ConfigOuterClass.Config;
029import hapi.chart.MetadataOuterClass.Maintainer;
030import hapi.chart.MetadataOuterClass.Metadata;
031import hapi.chart.TemplateOuterClass.Template;
032
033import org.yaml.snakeyaml.Yaml;
034
035public abstract class StreamOrientedChartLoader<T> implements ChartLoader<T> {
036
037  protected StreamOrientedChartLoader() {
038    super();
039  }
040  
041  protected Template createTemplate(final InputStream stream, final String name) throws IOException {
042    Template returnValue = null;
043    if (stream != null && name != null) {
044      final Template.Builder builder = Template.newBuilder();
045      builder.setName(name);
046      final ByteString data = ByteString.readFrom(stream);
047      assert data != null;
048      assert data.isValidUtf8();
049      builder.setData(data);
050      returnValue = builder.build();
051    }
052    return returnValue;
053  }
054  
055  protected Metadata createMetadata(final InputStream stream) throws IOException {
056    Metadata returnValue = null;
057    if (stream != null) {
058      final Map<?, ?> map = new Yaml().loadAs(stream, Map.class);
059      assert map != null;
060      final Metadata.Builder builder = Metadata.newBuilder();
061      assert builder != null;
062      @SuppressWarnings("unchecked")
063      final Iterable<String> keywords = (Iterable<String>)map.get("keywords");
064      if (keywords != null) {
065        builder.addAllKeywords(keywords);
066      }
067      @SuppressWarnings("unchecked")
068      final Iterable<? extends Map<?, ?>> maintainers = (Iterable<? extends Map<?, ?>>)map.get("maintainers");
069      if (maintainers != null) {
070        for (final Map<?, ?> maintainer : maintainers) {
071          if (maintainer != null) {
072            final Maintainer.Builder maintainerBuilder = Maintainer.newBuilder();
073            assert maintainerBuilder != null;
074            maintainerBuilder.setName((String)maintainer.get("name"));
075            maintainerBuilder.setEmail((String)maintainer.get("email"));
076            builder.addMaintainers(maintainerBuilder);
077          }
078        }
079      }
080      @SuppressWarnings("unchecked")
081      final Iterable<String> sources = (Iterable<String>)map.get("sources");
082      if (sources != null) {
083        builder.addAllSources(sources);
084      }
085      final String name = (String)map.get("name");
086      if (name != null) {
087        builder.setName(name);
088      }
089      final String version = (String)map.get("version");
090      if (version != null) {
091        builder.setVersion(version);
092      }
093      final String description = (String)map.get("description");
094      if (description != null) {
095        builder.setDescription(description);
096      }
097      final String engine = (String)map.get("engine");
098      if (engine != null) {
099        builder.setEngine(engine);
100      }
101      final String icon = (String)map.get("icon");
102      if (icon != null) {
103        builder.setIcon(icon);
104      }
105      final String appVersion = (String)map.get("appVersion");
106      if (appVersion != null) {
107        builder.setAppVersion(appVersion);
108      }
109      final String tillerVersion = (String)map.get("tillerVersion");
110      if (tillerVersion != null) {
111        builder.setTillerVersion(tillerVersion);
112      }
113      builder.setDeprecated("true".equals(String.valueOf(map.get("deprecated"))));
114      returnValue = builder.build();
115    }
116    return returnValue;
117  }
118  
119  protected Config createConfig(final InputStream stream) throws IOException {
120    Config returnValue = null;
121    if (stream != null) {
122      final Config.Builder builder = Config.newBuilder();
123      assert builder != null;
124      final ByteString rawBytes = ByteString.readFrom(stream);
125      assert rawBytes != null;
126      builder.setRawBytes(rawBytes);
127      returnValue = builder.build();
128    }
129    return returnValue;
130  }
131  
132  protected Any createAny(final InputStream stream, final String name) throws IOException {
133    Any returnValue = null;
134    if (stream != null && name != null) {
135      final Any.Builder builder = Any.newBuilder();
136      assert builder != null;
137      builder.setTypeUrl(name);
138      final ByteString fileContents = ByteString.readFrom(stream);
139      assert fileContents != null;
140      assert fileContents.isValidUtf8();
141      builder.setValue(fileContents);
142      returnValue = builder.build();
143    }
144    return returnValue;
145  }
146  
147}
