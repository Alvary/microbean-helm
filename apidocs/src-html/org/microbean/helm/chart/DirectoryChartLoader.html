055public class DirectoryChartLoader extends StreamOrientedChartLoader<Path> {
056
057  @Override
058  public Chart load(final Path path) throws IOException {
059    Chart returnValue = null;
060    if (path != null) {
061      if (Files.isDirectory(path)) {
062        returnValue = loadDirectory(path);
063      } else {
064        final URI pathUri = path.toUri();
065        assert pathUri != null;
066        final URL pathUrl = pathUri.toURL();
067        assert pathUrl != null;
068        try (final TarInputStream stream = new TarInputStream(new BufferedInputStream(new GZIPInputStream(pathUrl.openStream())))) {
069          returnValue = new TapeArchiveChartLoader().load(stream);
070        }
071      }
072    }
073    return returnValue;
074  }
075
076  private Chart loadDirectory(final Path path) throws IOException {
077    Chart returnValue = null;
078    if (path != null) {
079      if (Files.isDirectory(path)) {
080        final Chart[] chartHolder = new Chart[1];
081        Files.walkFileTree(path, new FileVisitor<Path>() {
082            
083            private Path startPath;
084            
085            private final List<Template> templates = new ArrayList<>();
086
087            private final List<Chart> subcharts = new ArrayList<>();
088
089            final List<Any> files = new ArrayList<>();
090
091            private Config config;
092
093            private Metadata metadata;
094
095            private ChartDirectoryType chartDirectoryType = ChartDirectoryType.NORMAL;
096
097            @Override
098            public FileVisitResult preVisitDirectory(final Path directory, final BasicFileAttributes attributes) throws IOException {
099              Objects.requireNonNull(directory);
100              Objects.requireNonNull(attributes);
101              assert attributes.isDirectory();
102              FileVisitResult returnValue = FileVisitResult.CONTINUE;
103              if (this.startPath == null) {
104                assert !isTemplateDirectory(directory);
105                assert !isSubchartDirectory(directory);
106                assert ChartDirectoryType.NORMAL.equals(this.chartDirectoryType);
107                this.startPath = directory;
108              } else if (ignore(directory)) {
109                returnValue = FileVisitResult.SKIP_SUBTREE;
110              } else if (isTemplateDirectory(directory)) {
111                // Entering "templates/".
112                this.chartDirectoryType = ChartDirectoryType.TEMPLATE;
113              } else if (isSubchartDirectory(directory)) {
114                // Entering "charts/".
115                this.chartDirectoryType = ChartDirectoryType.SUBCHART;
116              } else if (ChartDirectoryType.TEMPLATE.equals(this.chartDirectoryType)) {
117                // Entering a subdirectory of "templates/".  Carry on.
118              } else if (ChartDirectoryType.SUBCHART.equals(this.chartDirectoryType)) {
119                // Entering a subdirectory of "charts/".  Recursively
120                // load it and skip its entry.
121                this.subcharts.add(loadDirectory(directory));
122                returnValue = FileVisitResult.SKIP_SUBTREE;
123              } else {
124                // Entering a directory that is part of neither the
125                // "templates/" nor the "charts/" tree.  By definition
126                // it's normal.
127                this.chartDirectoryType = ChartDirectoryType.NORMAL;
128              }
129              return returnValue;
130            }
131
132            @Override
133            public FileVisitResult postVisitDirectory(final Path directory, final IOException exception) throws IOException {
134              if (exception != null) {
135                throw exception;
136              }
137              Objects.requireNonNull(directory);
138              this.chartDirectoryType = ChartDirectoryType.NORMAL;
139              if (directory != null && directory.equals(this.startPath)) {
140                assert chartHolder[0] == null;
141                final Chart.Builder chartBuilder = Chart.newBuilder();
142                assert chartBuilder != null;
143                if (this.metadata != null) {
144                  chartBuilder.setMetadata(this.metadata);
145                }
146                if (this.config != null) {
147                  chartBuilder.setValues(this.config);
148                }
149                chartBuilder.addAllTemplates(this.templates);
150                chartBuilder.addAllFiles(this.files);
151                chartBuilder.addAllDependencies(this.subcharts);
152                chartHolder[0] = chartBuilder.build();
153              }
154              return FileVisitResult.CONTINUE;
155            }
156            
157            @Override
158            public FileVisitResult visitFile(final Path file, final BasicFileAttributes attributes) throws IOException {
159              Objects.requireNonNull(file);
160              Objects.requireNonNull(attributes);
161              assert !attributes.isDirectory();
162              if (!ignore(file)) {
163
164                if (ChartDirectoryType.SUBCHART.equals(this.chartDirectoryType)) {
165                  // When we enter the charts/ directory, we do not
166                  // descend into it, but rather call ourselves
167                  // recursively (see above).  So we should never
168                  // encounter a file *inside* the charts/ directory
169                  // *as a subchart*.
170                  throw new IllegalStateException();
171                }
172                
173                assert this.startPath != null;
174
175                final Path relativePath = this.startPath.relativize(file);
176                assert relativePath != null;
177
178                try (final InputStream stream = new BufferedInputStream(file.toUri().toURL().openStream())) {
179                  if (ChartDirectoryType.TEMPLATE.equals(this.chartDirectoryType)) {
180                    this.templates.add(createTemplate(stream, relativePath.toString()));
181                  } else if (isValuesFile(file)) {
182                    if (this.config == null) {
183                      this.config = createConfig(stream);
184                    }
185                  } else if (isChartFile(file)) {
186                    if (this.metadata == null) {
187                      this.metadata = createMetadata(stream);
188                    }
189                  } else {
190                    this.files.add(createAny(stream, relativePath.toString()));
191                  }
192                }
193              }
194              return FileVisitResult.CONTINUE;
195            }
196
197            @Override
198            public FileVisitResult visitFileFailed(final Path file, final IOException exception) throws IOException {
199              if (exception != null) {
200                throw exception;
201              }
202              return FileVisitResult.CONTINUE;
203            }
204
205          });
206        returnValue = chartHolder[0];        
207      }
208    }
209    return returnValue;
210  }
211
212  protected boolean isChartFile(final Path path) {
213    return path != null && path.endsWith("Chart.yaml");
214  }
215  
216  protected boolean isValuesFile(final Path path) {
217    return path != null && path.endsWith("values.yaml");
218  }
219
220  protected boolean isSubchartDirectory(final Path path) {
221    return path != null && path.endsWith("charts");
222  }
223
224  protected boolean isTemplateDirectory(final Path path) {
225    return path != null && path.endsWith("templates");
226  }
227  
228  protected boolean ignore(final Path file) {
229    return false; // TODO: implement with .helmignore file, etc.
230  }
231
232  private static enum ChartDirectoryType {
233    NORMAL, SUBCHART, TEMPLATE
234  }
