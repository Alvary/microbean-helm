019import java.io.BufferedInputStream;
020import java.io.IOException;
021import java.io.InputStream;
022
023import java.net.URI;
024import java.net.URISyntaxException;
025import java.net.URL;
026
027import java.nio.file.Files;
028import java.nio.file.Path;
029import java.nio.file.Paths;
030
031import java.util.IdentityHashMap;
032import java.util.Collection;
033import java.util.Iterator;
034import java.util.Map.Entry;
035import java.util.Objects;
036
037import java.util.zip.GZIPInputStream;
038
039import hapi.chart.ChartOuterClass.Chart; // for javadoc only
040
041import org.kamranzafar.jtar.TarInputStream;
042
043/**
044 * A {@link StreamOrientedChartLoader StreamOrientedChartLoader&lt;URL&gt;} that creates
045 * {@link Chart} instances from {@link URL} instances.
046 *
047 * <h2>Thread Safety</h2>
048 *
049 * <p>This class is safe for concurrent use by multiple threads.</p>
050 *
051 * @author <a href="https://about.me/lairdnelson"
052 * target="_parent">Laird Nelson</a>
053 *
054 * @see #toNamedInputStreamEntries(URL)
055 *
056 * @see StreamOrientedChartLoader
057 */
058public class URLChartLoader extends StreamOrientedChartLoader<URL> {
059
060
061  /**
062   * Resources to be closed by the {@link #close()} method.
063   *
064   * <p>This field is never {@code null}.</p>
065   */
066  private final IdentityHashMap<AutoCloseable, Void> closeables;
067  
068
069  /*
070   * Constructors.
071   */
072
073  
074  /**
075   * Creates a new {@link URLChartLoader}.
076   */
077  public URLChartLoader() {
078    super();
079    this.closeables = new IdentityHashMap<>();
080  }
081
082
083  /*
084   * Instance methods.
085   */
086
087
088  /**
089   * Converts the supplied {@link URL} into an {@link Iterable} of
090   * {@link Entry} instances, each of which consists of an {@link
091   * InputStream} representing a resource within a Helm chart together
092   * with its (relative to the chart) name.
093   *
094   * <p>This method never returns {@code null}.</p>
095   *
096   * <p>Overrides of this method are not permitted to return {@code
097   * null}.
098   *
099   * @param url the {@link URL} to dereference; must be non-{@code
100   * null} or an effectively empty {@link Iterable} will be returned
101   *
102   * @return a non-{@code null} {@link Iterable} of {@link Entry}
103   * instances representing named {@link InputStream}s
104   *
105   * @exception IOException if there is a problem reading from the
106   * supplied {@link URL}
107   */
108  @Override
109  protected Iterable<? extends Entry<? extends String, ? extends InputStream>> toNamedInputStreamEntries(final URL url) throws IOException {
110    Objects.requireNonNull(url);
111    final String scheme = url.getProtocol();
112    Path path = null;
113    if ("file".equals(scheme)) {
114      URI uri = null;
115      try {
116        uri = url.toURI();
117      } catch (final URISyntaxException wrapMe) {
118        throw new IllegalArgumentException(wrapMe.getMessage(), wrapMe);
119      }
120      assert uri != null;
121      try {
122        path = Paths.get(uri);
123      } catch (final IllegalArgumentException notAFile) {
124        path = null;
125      }
126    }
127    final Iterable<? extends Entry<? extends String, ? extends InputStream>> returnValue;
128    if (path == null || !Files.isDirectory(path)) {
129      final TarInputStream tarInputStream = new TarInputStream(new GZIPInputStream(new BufferedInputStream(url.openStream())));
130      this.closeables.put(tarInputStream, null);
131      final TapeArchiveChartLoader loader = new TapeArchiveChartLoader();
132      this.closeables.put(loader, null);
133      returnValue = loader.toNamedInputStreamEntries(tarInputStream);
134    } else {
135      final DirectoryChartLoader loader = new DirectoryChartLoader();
136      this.closeables.put(loader, null);
137      returnValue = loader.toNamedInputStreamEntries(path);
138    }
139    return returnValue;
140  }
141
142  /**
143   * Closes resources opened by this {@link URLChartLoader}'s {@link
144   * #toNamedInputStreamEntries(URL)} method.
145   *
146   * @exception IOException if a subclass has overridden this method
147   * and an error occurs
148   */
149  @Override
150  public void close() throws Exception {
151    if (!this.closeables.isEmpty()) {
152      final Collection<? extends AutoCloseable> keys = this.closeables.keySet();
153      if (keys != null && !keys.isEmpty()) {
154        final Iterator<? extends AutoCloseable> iterator = keys.iterator();
155        if (iterator != null) {
156          while (iterator.hasNext()) {
157            final AutoCloseable closeable = iterator.next();
158            if (closeable != null) {
159              closeable.close();
160            }
161            iterator.remove();
162          }
163        }
164      }
165    }
166    assert this.closeables.isEmpty();
167  }
168  
169}
