041/**
042 * A {@link PathMatcher} and a {@link Predicate Predicate&lt;Path&gt;}
043 * that {@linkplain #matches(Path) matches} paths using the syntax of
044 * a {@code .helmignore} file.
045 *
046 * <p>This class passes <a
047 * href="https://github.com/kubernetes/helm/blob/v2.5.0/pkg/ignore/rules_test.go#L91-L121">all
048 * of the unit tests present</a> in the <a
049 * href="http://godoc.org/k8s.io/helm/pkg/ignore">Helm project's
050 * package concerned with {@code .helmignore} files</a>.  It may
051 * permit richer syntax, but there are no guarantees made regarding
052 * the behavior of this class in such cases.</p>
053 *
054 * <h2>Thread Safety</h2>
055 *
056 * <p>This class is safe for concurrent use by multiple threads.</p>
057 *
058 * @author <a href="https://about.me/lairdnelson"
059 * target="_parent">Laird Nelson</a>
060 *
061 * @see <a href="http://godoc.org/k8s.io/helm/pkg/ignore">The Helm
062 * project's package concerned with {@code .helmignore} files</a>
063 */
064public class HelmIgnorePathMatcher implements PathMatcher, Predicate<Path> {
065
066
067  /*
068   * Instance fields.
069   */
070
071
072  /**
073   * A {@link Collection} of {@link Predicate Predicate&lt;Path&gt;}s,
074   * one of which must {@linkplain #matches(Path) match} for the
075   * {@link #matches(Path)} method to return {@code true}.
076   *
077   * <p>This field is never {@code null}.</p>
078   *
079   * @see #addPatterns(Collection)
080   */
081  private final Collection<Predicate<Path>> rules;
082
083
084  /*
085   * Constructors.
086   */
087
088
089  /**
090   * Creates a new {@link HelmIgnorePathMatcher}.
091   */
092  public HelmIgnorePathMatcher() {
093    super();
094    this.rules = new ArrayList<>();
095    this.addPattern("templates/.?*");
096  }
097
098  /**
099   * Creates a new {@link HelmIgnorePathMatcher}.
100   *
101   * @param stringPatterns a {@link Collection} of <a
102   * href="http://godoc.org/k8s.io/helm/pkg/ignore">valid {@code
103   * .helmignore} patterns</a>; may be {@code null}
104   *
105   * @exception PatternSyntaxException if any of the patterns is
106   * invalid
107   */
108  public HelmIgnorePathMatcher(final Collection<? extends String> stringPatterns) {
109    this();
110    this.addPatterns(stringPatterns);
111  }
112
113  /**
114   * Creates a new {@link HelmIgnorePathMatcher}.
115   *
116   * @param reader a {@link Reader} expected to provide access to a
117   * logical collection of lines of text, each line of which is a <a
118   * href="http://godoc.org/k8s.io/helm/pkg/ignore">valid {@code
119   * .helmignore} pattern</a> (or blank line, or comment); may be
120   * {@code null}; never {@linkplain Reader#close() closed}
121   *
122   * @exception IOException if an error related to the supplied {@code
123   * reader} is encountered
124   *
125   * @exception PatternSyntaxException if any of the patterns is
126   * invalid
127   */
128  public HelmIgnorePathMatcher(final Reader reader) throws IOException {
129    this();
130    if (reader != null) {
131      final BufferedReader bufferedReader;
132      if (reader instanceof BufferedReader) {
133        bufferedReader = (BufferedReader)reader;
134      } else {
135        bufferedReader = new BufferedReader(reader);
136      }
137      assert bufferedReader != null;
138      final Collection<String> lines = new ArrayList<>();
139      String line = null;
140      while ((line = bufferedReader.readLine()) != null) {
141        lines.add(line);
142      }
143      if (!lines.isEmpty()) {
144        this.addPatterns(lines);
145      }
146    }
147  }
148
149  /**
150   * Creates a new {@link HelmIgnorePathMatcher}.
151   *
152   * @param helmIgnoreFile a {@link Path} expected to provide access
153   * to a logical collection of lines of text, each line of which is a
154   * <a href="http://godoc.org/k8s.io/helm/pkg/ignore">valid {@code
155   * .helmignore} pattern</a> (or blank line, or comment); may be
156   * {@code null}; never {@linkplain Reader#close() closed}
157   *
158   * @exception IOException if an error related to the supplied {@code
159   * helmIgnoreFile} is encountered
160   *
161   * @exception PatternSyntaxException if any of the patterns is
162   * invalid
163   *
164   * @see #HelmIgnorePathMatcher(Reader)
165   */
166  public HelmIgnorePathMatcher(final Path helmIgnoreFile) throws IOException {
167    this(Files.newBufferedReader(helmIgnoreFile));
168  }

172   * Instance methods.
173   */

176  /**
177   * Calls the {@link #addPatterns(Collection)} method with a
178   * {@linkplain Collections#singleton(Object) singleton
179   * <code>Set</code>} consisting of the supplied {@code
180   * stringPattern}.
181   *
182   * @param stringPattern a <a
183   * href="http://godoc.org/k8s.io/helm/pkg/ignore">valid {@code
184   * .helmignore} pattern</a>; may be {@code null} or {@linkplain
185   * String#isEmpty() empty} or prefixed with a {@code #} character,
186   * in which case no action will be taken
187   *
188   * @see #addPatterns(Collection)
189   *
190   * @see #matches(Path)
191   */
192  public final void addPattern(final String stringPattern) {
193    this.addPatterns(Collections.singleton(stringPattern));
194  }
195
196  /**
197   * Adds all of the <a
198   * href="http://godoc.org/k8s.io/helm/pkg/ignore">valid {@code
199   * .helmignore} patterns</a> present in the supplied {@link
200   * Collection} of such patterns.
201   *
202   * @param stringPatterns a {@link Collection} of <a
203   * href="http://godoc.org/k8s.io/helm/pkg/ignore">valid {@code
204   * .helmignore} patterns</a>; may be {@code null} in which case no
205   * action will be taken
206   *
207   * @see #matches(Path)
208   */
209  public void addPatterns(final Collection<? extends String> stringPatterns) {    
210    if (stringPatterns != null && !stringPatterns.isEmpty()) {
211      for (String stringPattern : stringPatterns) {
212        if (stringPattern != null && !stringPattern.isEmpty()) {
213          stringPattern = stringPattern.trim();
214          if (!stringPattern.isEmpty() && !stringPattern.startsWith("#")) {
215            if (stringPattern.contains("**")) {
216              throw new IllegalArgumentException("double-star (**) syntax is not supported"); // see rules.go
217            }
218
219            final boolean negate;
220            if (stringPattern.startsWith("!")) {
221              if (stringPattern.length() <= 1) {
222                throw new IllegalArgumentException(stringPattern);
223              }
224              negate = true;
225              stringPattern = stringPattern.substring(1);
226            } else {
227              negate = false;
228            }
229
230            final boolean requireDirectory;
231            if (stringPattern.endsWith("/")) {
232              if (stringPattern.length() <= 1) {
233                throw new IllegalArgumentException(stringPattern);
234              }
235              stringPattern = stringPattern.substring(0, stringPattern.length() - 1);
236              requireDirectory = true;
237            } else {
238              requireDirectory = false;
239            }
240
241            final boolean basename;
242            final int slashIndex = stringPattern.indexOf('/');
243            if (slashIndex == 0) {
244              if (stringPattern.length() <= 1) {
245                stringPattern = "";
246              } else {
247                stringPattern = stringPattern.substring(1);
248              }
249              basename = false;
250            } else if (slashIndex > 0) {
251              basename = false;
252            } else {
253              basename = true;
254            }
255
256            final StringBuilder regex = new StringBuilder();
257            regex.append("^"); // From Go's Filepath.Match: "Match requires pattern to match all of name, not just a substring."
258
259            final char[] chars = stringPattern.toCharArray();
260            assert chars != null;
261            assert chars.length > 0;
262            final int length = chars.length;
263            for (int i = 0; i < length; i++) {
264              final char c = chars[i];
265              switch (c) {
266              case '.':
267                regex.append("\\.");
268                break;
269              case '*':
270                // "matches any sequence of non-Separator characters"
271                regex.append("[^").append(File.separator).append("]*");
272                break;
273              case '?':
274                regex.append("[^").append(File.separator).append("]?");
275                break;
276              default:
277                regex.append(String.valueOf(c));
278                break;
279              }
280            }
281
282            regex.append("$"); // From Go's Filepath.Match: "Match requires pattern to match all of name, not just a substring."
283            final Pattern pattern = Pattern.compile(regex.toString());
284            final Predicate<Path> rule = new RegexRule(pattern, requireDirectory, basename);
285            synchronized (this.rules) {
286              this.rules.add(negate ? rule.negate() : rule);
287            }
288          }
289        }
290      }
291    }
292  }
293
294  /**
295   * Calls the {@link #matches(Path)} method with the supplied {@link
296   * Path} and returns its results.
297   *
298   * @param path a {@link Path} to test; may be {@code null}
299   *
300   * @return {@code true} if the supplied {@code path} matches; {@code
301   * false} otherwise
302   *
303   * @see #matches(Path)
304   */
305  @Override
306  public final boolean test(final Path path) {
307    return this.matches(path);
308  }
309
310  /**
311   * Returns {@code true} if at least one of the patterns added via
312   * the {@link #addPatterns(Collection)} method logically matches the
313   * supplied {@link Path}.
314   *
315   * @param path the {@link Path} to match; may be {@code null} in
316   * which case {@code false} will be returned
317   *
318   * @return {@code true} if at least one of the patterns added via
319   * the {@link #addPatterns(Collection)} method logically matches the
320   * supplied {@link Path}; {@code false} otherwise
321   */
322  @Override
323  public boolean matches(final Path path) {
324    boolean returnValue = false;
325    if (path != null) {
326      // https://github.com/kubernetes/helm/issues/1776
327      final String pathString = path.toString();
328      if (!pathString.equals(".") && !pathString.equals("./")) {
329        synchronized (this.rules) {
330          for (final Predicate<Path> rule : this.rules) {
331            if (rule != null && rule.test(path)) {
332              returnValue = true;
333              break;
334            }
335          }
336        }
337      }
338    }
339    return returnValue;
340  }

348  /**
349   * A {@link Predicate Predicate&lt;Path&gt;} that may also apply
350   * {@link Path}-specific tests.
351   *
352   * @author <a href="https://about.me/lairdnelson"
353   * target="_parent">Laird Nelson</a>
354   */
355  private static abstract class Rule implements Predicate<Path> {
356
357
358    /*
359     * Instance fields.
360     */
361
362
363    /**
364     * Whether a {@link Path} must {@linkplain Files#isDirectory(Path)
365     * be a directory} in order for this {@link Rule} to match.
366     */
367    private final boolean requireDirectory;
368
369    /**
370     * Whether the {@linkplain Path#getFileName() final component in a
371     * <code>Path</code>} is matched, or the entire {@link Path}.
372     */
373    private final boolean basename;
374
375
376    /*
377     * Constructors.
378     */
379
380
381    /**
382     * Creates a new {@link Rule}.
383     *
384     * @param requireDirectory whether a {@link Path} must {@linkplain
385     * Files#isDirectory(Path) be a directory} in order for this
386     * {@link Rule} to match
387     *
388     * @param basename hhether the {@linkplain Path#getFileName()
389     * final component in a <code>Path</code>} is matched, or the
390     * entire {@link Path}
391     */
392    protected Rule(final boolean requireDirectory, final boolean basename) {
393      super();
394      this.requireDirectory = requireDirectory;
395      this.basename = basename;
396    }
397
398    /**
399     * Returns a {@link Path} that can be tested, given a {@link Path}
400     * and the application of the {@code requireDirectory} and {@code
401     * basename} parameters passed to the {@linkplain #Rule(boolean,
402     * boolean) constructor}.
403     *
404     * <p>This method may return {@code null}.</p>
405     *
406     * @param path the {@link Path} to normalize; may be {@code null}
407     * in which case {@code null} will be returned
408     *
409     * @return a {@link Path} to be further tested; or {@code null}
410     */
411    protected final Path normalizePath(final Path path) {
412      Path returnValue = path;
413      if (path != null) {
414        if (this.basename) {
415          returnValue = path.getFileName();
416        }
417        if (this.requireDirectory && !Files.isDirectory(path)) {
418          returnValue = null;
419        }
420      }
421      return returnValue;
422    }
423
424  }
425
426  /**
427   * A {@link Rule} that uses regular expressions to match {@link Path}s.
428   *
429   * @author <a href="https://about.me/lairdnelson"
430   * target="_parent">Laird Nelson</a>
431   *
432   * @see Pattern
433   */
434  private static final class RegexRule extends Rule {
435
436
437    /*
438     * Instance fields.
439     */
440
441
442    /**
443     * The {@link Pattern} specifying what {@link Path} instances
444     * should be matched.
445     *
446     * <p>This field may be {@code null}.</p>
447     */
448    private final Pattern pattern;
449
450
451    /*
452     * Constructors.
453     */
454
455
456    /**
457     * Creates a new {@link RegexRule}.
458     *
459     * @param pattern the {@link Pattern} specifying what {@link Path}
460     * instances should be matched; may be {@code null}
461     *
462     * @param requireDirectory whether only {@link Path} instances
463     * that {@linkplain Files#isDirectory(Path) are directories} are
464     * subject to further matching
465     *
466     * @param basename whether only {@linkplain Path#getFileName() the
467     * last component of a <code>Path</code>} is considered for
468     * matching
469     *
470     * @see #test(Path)
471     */
472    private RegexRule(final Pattern pattern, final boolean requireDirectory, final boolean basename) {
473      super(requireDirectory, basename);
474      this.pattern = pattern;
475    }
476
477
478    /*
479     * Instance methods.
480     */
481
482
483    /**
484     * Tests the supplied {@link Path} to see if it matches the
485     * conditions supplied at construction time.
486     *
487     * @param path the {@link Path} to test; may be {@code null} in
488     * which case {@code false} will be returned
489     *
490     * @return {@code true} if this {@link RegexRule} matches the
491     * supplied {@link Path}; {@code false} otherwise
492     */
493    @Override
494    public final boolean test(Path path) {
495      boolean returnValue = false;
496      path = this.normalizePath(path);
497      if (path != null) {
498        if (this.pattern == null) {
499          returnValue = true;
500        } else {
501          final Matcher matcher = this.pattern.matcher(path.toString());
502          assert matcher != null;
503          returnValue = matcher.matches();
504        }
505      }
506      return returnValue;
507    }
508  }
