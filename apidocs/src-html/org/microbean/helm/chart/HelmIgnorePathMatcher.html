017package org.microbean.helm.chart;
018
019import java.io.BufferedReader;
020import java.io.File;
021import java.io.IOException;
022import java.io.Reader;
023
024import java.nio.file.Files;
025import java.nio.file.Path;
026
027import java.nio.file.PathMatcher;
028
029import java.util.ArrayList;
030import java.util.Collection;
031import java.util.Collections;
032
033import java.util.function.Predicate;
034
035import java.util.regex.Matcher;
036import java.util.regex.Pattern;
037
038import java.util.stream.Collectors;
039
040public class HelmIgnorePathMatcher implements PathMatcher, Predicate<Path> {
041
042  private final Collection<Rule> rules;
043  
044  public HelmIgnorePathMatcher() {
045    super();
046    this.rules = new ArrayList<>();
047  }
048
049  public HelmIgnorePathMatcher(final Collection<? extends String> stringPatterns) {
050    this();
051    this.addPatterns(stringPatterns);
052  }
053
054  public HelmIgnorePathMatcher(final Reader reader) throws IOException {
055    this();
056    if (reader != null) {
057      final BufferedReader bufferedReader;
058      if (reader instanceof BufferedReader) {
059        bufferedReader = (BufferedReader)reader;
060      } else {
061        bufferedReader = new BufferedReader(reader);
062      }
063      assert bufferedReader != null;
064      final Collection<String> lines = new ArrayList<>();
065      String line = null;
066      while ((line = bufferedReader.readLine()) != null) {
067        lines.add(line);
068      }
069      if (!lines.isEmpty()) {
070        this.addPatterns(lines);
071      }
072    }
073  }
074  
075  public HelmIgnorePathMatcher(final Path helmIgnoreFile) throws IOException {
076    this();
077    if (helmIgnoreFile != null) {
078      final Collection<? extends String> lines = Files.lines(helmIgnoreFile).collect(Collectors.toList());
079      if (lines != null && !lines.isEmpty()) {
080        this.addPatterns(lines);
081      }
082    }
083  }
084
085  public void addPattern(final String stringPattern) {
086    this.addPatterns(Collections.singleton(stringPattern));
087  }
088  
089  public void addPatterns(final Collection<? extends String> stringPatterns) {
090    if (stringPatterns != null && !stringPatterns.isEmpty()) {
091      for (String stringPattern : stringPatterns) {
092        if (stringPattern != null && !stringPattern.isEmpty()) {
093          stringPattern = stringPattern.trim();
094          if (!stringPattern.isEmpty() && !stringPattern.startsWith("#")) {
095            if (stringPattern.contains("**")) {
096              throw new IllegalArgumentException("double-star (**) syntax is not supported"); // see rules.go
097            }
098
099            final boolean negate;
100            if (stringPattern.startsWith("!")) {
101              if (stringPattern.length() <= 1) {
102                throw new IllegalArgumentException(stringPattern);
103              }
104              negate = true;
105              stringPattern = stringPattern.substring(1);
106            } else {
107              negate = false;
108            }
109
110            final boolean requireDirectory;
111            if (stringPattern.endsWith("/")) {
112              if (stringPattern.length() <= 1) {
113                throw new IllegalArgumentException(stringPattern);
114              }
115              stringPattern = stringPattern.substring(0, stringPattern.length() - 1);
116              requireDirectory = true;
117            } else {
118              requireDirectory = false;
119            }
120
121            final boolean basename;
122            final int slashIndex = stringPattern.indexOf('/');
123            if (slashIndex == 0) {
124              if (stringPattern.length() <= 1) {
125                stringPattern = "";
126              } else {
127                stringPattern = stringPattern.substring(1);
128              }
129              basename = false;
130            } else if (slashIndex > 0) {
131              basename = false;
132            } else {
133              basename = true;
134            }
135
136            final StringBuilder regex = new StringBuilder();
137            // regex.append("^"); // From Go's Filepath.Match: "Match requires pattern to match all of name, not just a substring."
138
139            final char[] chars = stringPattern.toCharArray();
140            assert chars != null;
141            assert chars.length > 0;
142            final int length = chars.length;
143            for (int i = 0; i < length; i++) {
144              final char c = chars[i];
145              switch (c) {
146              case '.':
147                regex.append("\\.");
148                break;
149              case '*':
150                // "matches any sequence of non-Separator characters"
151                regex.append("[^").append(File.separator).append("]*");
152                break;
153              case '?':
154                regex.append("[^").append(File.separator).append("]");
155                break;
156              case '\\':
157                if (i + 1 == length) {
158                  throw new IllegalArgumentException(stringPattern);
159                }
160                regex.append(String.valueOf(chars[++i]));
161                break;
162              default:
163                regex.append(String.valueOf(c));
164                break;
165              }
166            }
167
168            // regex.append("$"); // From Go's Filepath.Match: "Match requires pattern to match all of name, not just a substring."
169            final Pattern pattern = Pattern.compile(regex.toString());
170            this.rules.add(new Rule(pattern, negate, requireDirectory, basename));
171          }
172        }
173      }
174    }
175  }
176  
177  @Override
178  public boolean test(final Path path) {
179    return this.matches(path);
180  }
181  
182  @Override
183  public boolean matches(final Path path) {
184    if (path != null) {
185      // https://github.com/kubernetes/helm/issues/1776
186      final String pathString = path.toString();
187      if (pathString.equals(".") || pathString.equals("./")) {
188        return false;
189      }
190    }
191    for (final Rule rule : this.rules) {
192      if (rule != null && rule.test(path)) {
193        return true;
194      }
195    }
196    return false;
197  }
198  
199  private static final class Rule implements Predicate<Path> {
200
201    private final Pattern pattern;
202
203    private final boolean negate;
204
205    private final boolean requireDirectory;
206
207    private final boolean basename;
208    
209    private Rule(final Pattern pattern, final boolean negate, final boolean requireDirectory, final boolean basename) {
210      super();
211      this.pattern = pattern;
212      this.negate = negate;
213      this.requireDirectory = requireDirectory;
214      this.basename = basename;
215    }
216
217    @Override
218    public boolean test(Path path) {
219      boolean returnValue = false;
220      if (path != null) {
221        if (this.basename) {
222          path = path.getFileName();
223        }
224        
225        if (!this.requireDirectory || Files.isDirectory(path)) {
226          if (this.pattern == null) {
227            returnValue = true;
228          } else {
229            final String pathString = path.toString();
230            assert pathString != null;
231            final Matcher matcher = this.pattern.matcher(pathString);
232            assert matcher != null;
233            returnValue = matcher.matches();
234          }
235        }
236      }
237      if (this.negate) {
238        returnValue = !returnValue;
239      }
240      return returnValue;
241    }
242    
243  }
