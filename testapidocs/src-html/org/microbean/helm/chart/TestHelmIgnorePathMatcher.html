054@RunWith(Parameterized.class)
055public class TestHelmIgnorePathMatcher {
056
057  @Parameters(name = "{0}, {1}, {2}")
058  public static Collection<Object[]> data() {
059    return Arrays.asList(new Object[][] {
060        {"helm.txt", "helm.txt", true},
061        {"helm.*", "helm.txt", true},
062        {"helm.*", "rudder.txt", false},
063        {"*.txt", "tiller.txt", true},
064        {"*.txt", "cargo/a.txt", true},
065        {"cargo/*.txt", "cargo/a.txt", true},
066        {"cargo/*.*", "cargo/a.txt", true},
067        {"cargo/*.txt", "mast/a.txt", false},
068        {"ru[c-e]?er.txt", "rudder.txt", true},
069        {"templates/.?*", "templates/.dotfile", true},
070        // "." should never get ignored. https://github.com/kubernetes/helm/issues/1776
071        {".*", ".", false},
072        {".*", "./", false},
073
074        
075        // Directory tests
076        {"cargo/", "cargo", true},
077        {"cargo/", "cargo/", true},
078        {"cargo/", "mast/", false},
079        {"helm.txt/", "helm.txt", false},
080        
081        // Negation tests
082        {"!helm.txt", "helm.txt", false},
083        {"!helm.txt", "tiller.txt", true},
084        {"!*.txt", "cargo", true},
085        {"!cargo/", "mast/", true},
086        
087        // Absolute path tests
088        {"/a.txt", "a.txt", true},
089        {"/a.txt", "cargo/a.txt", false},
090        {"/cargo/a.txt", "cargo/a.txt", true}
091      });
092  }
093
094  private final String pattern;
095
096  private final String testText;
097
098  private final boolean match;
099
100  private HelmIgnorePathMatcher pathMatcher;
101  
102  public TestHelmIgnorePathMatcher(final String pattern, final String testText, final boolean match) {
103    super();
104    this.pattern = pattern;
105    this.testText = testText;
106    this.match = match;
107  }
108
109  @Before
110  public void setUp() {
111    this.pathMatcher = new HelmIgnorePathMatcher();
112  }
113
114  @After
115  public void tearDown() {
116
117  }
118
119  @Test
120  public void testStuff() {
121    this.pathMatcher.addPattern(this.pattern);
122    assertEquals(this.pattern + " did not match " + this.testText, this.match, this.pathMatcher.matches(Paths.get(this.testText)));
123  }
