088  @Test
089  public void testYamlUnmarshalling() throws IOException {
090    final URL requirementsYaml = Thread.currentThread().getContextClassLoader().getResource("TestRequirements/requirements.yaml");
091    assertNotNull(requirementsYaml);
092    try (final InputStream stream = new BufferedInputStream(requirementsYaml.openStream())) {
093      final Yaml yaml = new Yaml();
094      final Requirements requirements = yaml.loadAs(stream, Requirements.class);
095      assertNotNull(requirements);
096      final Collection<Dependency> dependencies = requirements.getDependencies();
097      assertNotNull(dependencies);
098      assertEquals(2, dependencies.size());
099      final Iterator<Dependency> iterator = dependencies.iterator();
100      assertNotNull(iterator);
101      assertTrue(iterator.hasNext());
102
103      final Dependency firstDependency = iterator.next();
104      assertEquals("subchart1", firstDependency.getName());
105      assertEquals("http://localhost:10191", firstDependency.getRepository());
106      assertEquals("0.1.0", firstDependency.getVersion());
107      assertEquals("subchart1.enabled, global.subchart1.enabled", firstDependency.getCondition());
108      Collection<String> tags = firstDependency.getTags();
109      assertNotNull(tags);
110      assertEquals(2, tags.size());
111      
112      assertTrue(iterator.hasNext());
113      
114      final Dependency secondDependency = iterator.next();
115      assertEquals("subchart2", secondDependency.getName());
116      assertEquals("http://localhost:10191", secondDependency.getRepository());
117      assertEquals("0.1.0", secondDependency.getVersion());
118      assertEquals("subchart2.enabled,global.subchart2.enabled", secondDependency.getCondition());
119      tags = secondDependency.getTags();
120      assertNotNull(tags);
121      assertEquals(2, tags.size());
122
123      assertFalse(iterator.hasNext());
124    }
125  }

182  private final void verifyRequirementsEnabled(final String tags, final String... expectedChartNames) {
183    assertNotNull(tags);

185    final Config.Builder configBuilder = Config.newBuilder();
186    assertNotNull(configBuilder);

189    for (final String expectation : expectedChartNames) {

261  private final void verifyRequirementsEnabled(final ConfigOrBuilder config, final SortedSet<? extends String> expectations) {
262    assertNotNull(this.chartBuilder);
263    assertNotNull(config);
264    assertNotNull(expectations);
265
266    assertSame(this.chartBuilder, Requirements.apply(this.chartBuilder, config, false));
267
268    assertEquals(expectations, getChartNames());
269  }
270
271  private final SortedSet<? extends String> getChartNames() {
272    final SortedSet<String> returnValue = new TreeSet<>();
273    
274    final List<ChartOrBuilder> charts = new LinkedList<>();
275    charts.add(this.chartBuilder);
276    
277    while (!charts.isEmpty()) {
278      ChartOrBuilder chart = charts.remove(0);
279      assertNotNull(chart);
280      final MetadataOrBuilder metadata = chart.getMetadataOrBuilder();
281      if (metadata != null) {
282        final String name = metadata.getName();
283        if (name != null && !name.isEmpty()) {
284          returnValue.add(name);
285        }
286      }
287      final Collection<? extends ChartOrBuilder> subcharts = chart.getDependenciesOrBuilderList();
288      if (subcharts != null && !subcharts.isEmpty()) {
289        charts.addAll(subcharts);
290      }
291    }
292    return returnValue;
293  }

299  private final void verifyRequirementsImportValues(final ConfigOrBuilder config, final Map<? extends String, ? extends String> expectations) {
300    assertNotNull(this.chartBuilder);
301    final Requirements requirements = Requirements.fromChartOrBuilder(this.chartBuilder);
302    assertNotNull(requirements);
303    assertSame(this.chartBuilder, Requirements.processImportValues(this.chartBuilder));
304    final Map<String, Object> chartDefaultValues = Configs.toDefaultValuesMap(this.chartBuilder);
305    assertNotNull(chartDefaultValues);
306    final MapTree defaultValues = new MapTree(chartDefaultValues);
307    final Set<? extends Entry<? extends String, ? extends String>> expectationsEntrySet = expectations.entrySet();
308    assertNotNull(expectationsEntrySet);
309    for (final Entry<? extends String, ? extends String> expectationsEntry : expectationsEntrySet) {
310      assertNotNull(expectationsEntry);
311      final String path = expectationsEntry.getKey();
312      assertNotNull(path);
313      final Object pathValue = defaultValues.get(path, Object.class);
314      assertNotNull(pathValue);
315      if (path.endsWith("bool")) {
316        assertTrue(pathValue instanceof Boolean);
317      } else if (path.endsWith("string")) {
318        assertTrue(pathValue instanceof String);
319      } else if (path.endsWith("float")) {
320        assertTrue(pathValue instanceof Double); // Java/Go type mismatch
321      } else if (path.endsWith("int")) {
322        assertTrue(pathValue instanceof Integer);
323      }
324      final Object expectedValue = expectationsEntry.getValue();
325      assertNotNull(expectedValue);
326      assertEquals(expectedValue.toString(), pathValue.toString());
327    }
328  }
