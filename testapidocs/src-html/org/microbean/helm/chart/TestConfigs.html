017package org.microbean.helm.chart;
018
019import java.io.IOException;
020
021import java.net.URL;
022
023import java.util.Map;
024
025import hapi.chart.ChartOuterClass.Chart;
026import hapi.chart.ChartOuterClass.ChartOrBuilder;
027
028import org.junit.After;
029import org.junit.Before;
030import org.junit.Test;
031
032import org.yaml.snakeyaml.Yaml;
033
034import static org.junit.Assert.assertEquals;
035import static org.junit.Assert.assertNotNull;
036
037public class TestConfigs {
038
039  private AbstractChartLoader<URL> chartLoader;
040  
041  private Chart chart;
042 
043  public TestConfigs() {
044    super();
045  }
046
047  @Before
048  public void setUp() throws IOException {
049    final URL chartLocation = Thread.currentThread().getContextClassLoader().getResource("TestConfigs/subpop");
050    assertNotNull(chartLocation);
051    this.chartLoader = new URLChartLoader();
052    this.chart = this.chartLoader.load(chartLocation).build();
053    assertNotNull(this.chart);
054  }
055
056  @After
057  public void tearDown() throws IOException {
058    if (this.chartLoader != null) {
059      this.chartLoader.close();
060    }
061  }
062  
063  @Test
064  public void testToDefaultValuesMap() {
065    final Map<String, Object> values = Configs.toDefaultValuesMap(this.chart);
066    testDefaultValuesMap(values);
067  }
068
069  @Test
070  public void testToValuesMapWithNullAsSecondArgument() {
071    final Map<String, Object> defaultValues = Configs.toDefaultValuesMap(this.chart);
072    testDefaultValuesMap(defaultValues);
073    final Map<String, Object> shouldBeDefaultValuesToo = Configs.toValuesMap(this.chart, null);
074    assertEquals(defaultValues, shouldBeDefaultValuesToo);
075  }
076
077  private static final void testDefaultValuesMap(final Map<? extends String, ?> values) {
078    assertNotNull(values);
079    assertEquals(9, values.size());
080
081    @SuppressWarnings("unchecked")
082    final Map<? extends String, ?> importedChart1 = (Map<? extends String, ?>)values.get("imported-chart1");
083    assertNotNull(importedChart1);
084    assertEquals(1, importedChart1.size());
085    assertEquals("helm rocks", importedChart1.get("SPextra1"));
086
087    @SuppressWarnings("unchecked")
088    final Map<? extends String, ?> overriddenChart1 = (Map<? extends String, ?>)values.get("overridden-chart1");
089    assertNotNull(overriddenChart1);
090    assertEquals(5, overriddenChart1.size());
091    assertEquals(Boolean.FALSE, overriddenChart1.get("SC1bool"));
092    assertEquals(Double.valueOf(3.141592d), overriddenChart1.get("SC1float"));
093    assertEquals(Integer.valueOf(99), overriddenChart1.get("SC1int"));
094    assertEquals("pollywog", overriddenChart1.get("SC1string"));
095    assertEquals(Integer.valueOf(42), overriddenChart1.get("SPextra2"));
096
097    @SuppressWarnings("unchecked")
098    final Map<? extends String, ?> importedChartA = (Map<? extends String, ?>)values.get("imported-chartA");
099    assertNotNull(importedChartA);
100    assertEquals(1, importedChartA.size());
101    assertEquals(Double.valueOf(1.337d), importedChartA.get("SPextra3"));
102
103    @SuppressWarnings("unchecked")
104    final Map<? extends String, ?> overriddenChartA = (Map<? extends String, ?>)values.get("overridden-chartA");
105    assertNotNull(overriddenChartA);
106    assertEquals(5, overriddenChartA.size());
107    assertEquals(Boolean.TRUE, overriddenChartA.get("SCAbool"));
108    assertEquals(Double.valueOf(41.3), overriddenChartA.get("SCAfloat"));
109    assertEquals(Integer.valueOf(808), overriddenChartA.get("SCAint"));
110    assertEquals("jaberwocky", overriddenChartA.get("SCAstring"));
111    assertEquals(Boolean.TRUE, overriddenChartA.get("SPextra4"));
112    
113    @SuppressWarnings("unchecked")
114    final Map<? extends String, ?> importedChartAB = (Map<? extends String, ?>)values.get("imported-chartA-B");
115    assertNotNull(importedChartAB);
116    assertEquals(1, importedChartAB.size());
117    assertEquals("k8s", importedChartAB.get("SPextra5"));
118
119    @SuppressWarnings("unchecked")
120    final Map<? extends String, ?> overriddenChartAB = (Map<? extends String, ?>)values.get("overridden-chartA-B");
121    assertNotNull(overriddenChartAB);
122    assertEquals(9, overriddenChartAB.size());
123    assertEquals(Boolean.TRUE, overriddenChartAB.get("SCAbool"));
124    assertEquals(Double.valueOf(41.3), overriddenChartAB.get("SCAfloat"));
125    assertEquals(Integer.valueOf(808), overriddenChartAB.get("SCAint"));
126    assertEquals("jaberwocky", overriddenChartAB.get("SCAstring"));
127    assertEquals(Boolean.FALSE, overriddenChartAB.get("SCBbool"));
128    assertEquals(Double.valueOf(1.99), overriddenChartAB.get("SCBfloat"));
129    assertEquals(Integer.valueOf(77), overriddenChartAB.get("SCBint"));
130    assertEquals("jango", overriddenChartAB.get("SCBstring"));
131    assertEquals(Integer.valueOf(111), overriddenChartAB.get("SPextra6"));
132    
133    @SuppressWarnings("unchecked")
134    final Map<? extends String, ?> tags = (Map<? extends String, ?>)values.get("tags");
135    assertNotNull(tags);
136    assertEquals(2, tags.size());
137    assertEquals(Boolean.TRUE, tags.get("front-end"));
138    assertEquals(Boolean.FALSE, tags.get("back-end"));
139
140    System.out.println(new Yaml().dump(values));
141  }
